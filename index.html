/* ------------------ Config & weights ------------------ */

const EXAMS = {

  Jan: new Date('2026-01-12T00:00:00'),

  Jul: new Date('2026-07-13T00:00:00')

};

// percentages you provided

const FLK1 = [

  {key:'Business Law & Practice', pct:10.5},

  {key:'Dispute Resolution', pct:10.5},

  {key:'Contract Law', pct:5.3},

  {key:'Tort Law', pct:5.3},

  {key:'Constitutional & Administrative Law', pct:5.3},

  {key:'Legal System of England & Wales', pct:5.3},

  {key:'Legal Services / SRA Regulation', pct:5.3},

  {key:'Solicitors’ Accounts', pct:5.3},

  {key:'Ethics & Taxation', pct:10.5},

];

const FLK2 = [

  {key:'Property Practice', pct:5.3},

  {key:'Land Law', pct:5.3},

  {key:'Wills & Administration of Estates', pct:5.3},

  {key:'Trusts', pct:10.5},

  {key:'Criminal Law & Practice', pct:10.5},

];

function weightsFor(stream){

  if(stream==='FLK1') return FLK1;

  if(stream==='FLK2') return FLK2;

  return [...FLK1, ...FLK2];

}

 

/* ------------------ Utilities ------------------ */

const $ = s=>document.querySelector(s);

function addDays(d, n){ const x=new Date(d); x.setDate(x.getDate()+n); return x; }

function startOfDay(d){ const x=new Date(d); x.setHours(0,0,0,0); return x; }

function fmtDate(d){

  return d.toLocaleDateString(undefined,{day:'2-digit',month:'short',year:'numeric'});

}

function weekRange(start){

  const end = addDays(start,6);

  return `${fmtDate(start)} – ${fmtDate(end)}`;

}

function weeksBetweenInclusiveStartExclusiveEnd(start, end){

  const ms = startOfDay(end) - startOfDay(start);

  const days = Math.floor(ms / (1000*60*60*24));

  return Math.floor(days/7);

}

 

/* ------------------ Planning logic ------------------ */

function allocateWeeks(totalWeeks, stream){

  const items = weightsFor(stream).map(t=>({topic:t.key, pct:t.pct}));

  // Convert percentages to shares out of planning weeks

  const shares = items.map(it=>{

    const raw = totalWeeks * (it.pct/100);

    return {topic:it.topic, raw, floor:Math.floor(raw), rem: raw - Math.floor(raw)};

  });

  let assigned = shares.reduce((s,a)=>s+a.floor,0);

  let leftover = totalWeeks - assigned;

  // Distribute leftover by largest remainder

  shares.sort((a,b)=>b.rem-a.rem);

  for(let i=0;i<leftover;i++){ shares[i % shares.length].floor++; }

  // Return counts

  shares.sort((a,b)=> a.topic.localeCompare(b.topic)); // stable order later not important

  return shares.map(s=>({topic:s.topic, weeks:s.floor})).filter(x=>x.weeks>0);

}

 

function buildSequence(startDate, examDate, stream){

  const total = weeksBetweenInclusiveStartExclusiveEnd(startDate, examDate);

  if(total < 1) return {error:"Not enough time: fewer than 1 full week. Choose a later sitting."};

  // revision rule

  const hasRevision = total >= 13;

  const reservedRevision = hasRevision ? 2 : 0;

  const planningWeeks = total - reservedRevision;

  if(planningWeeks < 1) return {error:"Planning window is too short after applying the revision rule."};

 

  // topic allocation

  const alloc = allocateWeeks(planningWeeks, stream);

 

  // Turn counts into contiguous blocks (simple sensible sequence)

  // Heuristic: interleave heavier blocks, place single-week topics between heavier ones.

  const singles = alloc.filter(a=>a.weeks===1);

  const multis = alloc.filter(a=>a.weeks>1).sort((a,b)=>b.weeks-a.weeks);

 

  const sequence = [];

  // place multis first as blocks

  multis.forEach(m=>{

    for(let w=1; w<=m.weeks; w++){

      sequence.push({type:'topic', topic:m.topic, part:w, of:m.weeks});

    }

  });

  // insert singles between heavier areas

  let insertIdx = 1;

  singles.forEach(s=>{

    const idx = Math.min(insertIdx, sequence.length);

    sequence.splice(idx,0,{type:'topic', topic:s.topic, part:1, of:1});

    insertIdx += 2;

  });

 

  // Insert revision at midpoint and end if applicable

  if(hasRevision){

    const mid = Math.floor(sequence.length/2);

    sequence.splice(mid,0,{type:'revision', title:'Revision Week 1 — mocks, timing practice, weak-area drills'});

    sequence.push({type:'revision', title:'Revision Week 2 — full-course review, logistics, rest'});

  }

 

  // Convert to week objects with calendar dates

  const weeks = [];

  let cursor = startOfDay(startDate);

  for(let i=0;i<sequence.length;i++){

    const s = sequence[i];

    const week = { index:i, start:new Date(cursor), isRevision:false };

    if(s.type==='topic'){

      week.title = s.of>1 ? `${s.topic} — Week ${s.part}` : s.topic;

      week.description = `Target: ${stream}. Prepare for ${examDate.toLocaleDateString(undefined,{month:'long',year:'numeric'})}. Focus on core subtopics and timed practice.`;

      week.topic = s.topic;

    }else{

      week.title = s.title;

      week.description = `Target: ${stream}. Prepare for ${examDate.toLocaleDateString(undefined,{month:'long',year:'numeric'})}.`;

      week.isRevision = true;

    }

    weeks.push(week);

    cursor = addDays(cursor,7);

  }

  return {weeks, total, hasRevision, planningWeeks};

}

 

/* ------------------ Table rendering & interaction ------------------ */

let MODEL = {weeks:[], startDate:null, examKey:'Jan', stream:'Both'};

 

const tbody = $('#tbody');

function render(){

  // meta

  const examDate = EXAMS[MODEL.examKey];

  $('#meta').textContent = `Start: ${fmtDate(MODEL.startDate)} · Exam week: ${fmtDate(examDate)} · Weeks: ${MODEL.weeks.length}`;

  $('#revBadge').hidden = !MODEL.weeks.some(w=>w.isRevision);

 

  // rows

  tbody.innerHTML = '';

  MODEL.weeks.forEach((w,i)=>{

    const tr = document.createElement('tr');

    if(w.isRevision) tr.classList.add('revision');

    tr.draggable = true;

    tr.dataset.index = i;

 

    tr.innerHTML = `

      <td class="handle" title="Drag to reorder" aria-label="Drag handle">⠿</td>

      <td><div class="week-title">${weekRange(w.start)}</div></td>

      <td><div class="week-title">${escapeHtml(w.title)}</div></td>

      <td>

        <details>

          <summary>Details</summary>

          <div class="muted"> ${escapeHtml(w.description)} </div>

          ${w.topic ? `<div class="muted" style="margin-top:6px">Topic: ${escapeHtml(w.topic)}</div>`:''}

        </details>

      </td>

      <td class="row-actions">

        <button class="btn" data-act="swapUp">↑ Swap</button>

        <button class="btn" data-act="swapDown">↓ Swap</button>

        <button class="btn" data-act="split">⤿ Split</button>

      </td>

    `;

    // drag events

    tr.addEventListener('dragstart', onDragStart);

    tr.addEventListener('dragover', onDragOver);

    tr.addEventListener('drop', onDrop);

    tr.addEventListener('dragend', onDragEnd);

 

    tbody.appendChild(tr);

  });

}

 

let dragIdx=null;

function onDragStart(e){

  const tr = e.currentTarget;

  dragIdx = parseInt(tr.dataset.index,10);

  tr.classList.add('dragging');

  e.dataTransfer.effectAllowed = 'move';

}

function onDragOver(e){

  e.preventDefault();

  e.dataTransfer.dropEffect = 'move';

}

function onDrop(e){

  e.preventDefault();

  const targetTr = e.currentTarget;

  const toIdx = parseInt(targetTr.dataset.index,10);

  moveRow(dragIdx, toIdx);

}

function onDragEnd(e){

  e.currentTarget.classList.remove('dragging');

}

function moveRow(from, to){

  if(from===to) return;

  const item = MODEL.weeks.splice(from,1)[0];

  MODEL.weeks.splice(to,0,item);

  recomputeDates();

  render();

}

 

tbody.addEventListener('click', (e)=>{

  const btn = e.target.closest('button');

  if(!btn) return;

  const tr = e.target.closest('tr');

  const idx = parseInt(tr.dataset.index,10);

  const act = btn.dataset.act;

 

  if(act==='swapUp' && idx>0){

    [MODEL.weeks[idx-1], MODEL.weeks[idx]] = [MODEL.weeks[idx], MODEL.weeks[idx-1]];

    recomputeDates(); render();

  }

  if(act==='swapDown' && idx<MODEL.weeks.length-1){

    [MODEL.weeks[idx+1], MODEL.weeks[idx]] = [MODEL.weeks[idx], MODEL.weeks[idx+1]];

    recomputeDates(); render();

  }

  if(act==='split'){

    splitWeek(idx);

  }

});

 

function splitWeek(idx){

  const w = MODEL.weeks[idx];

  // Don't split revision weeks; silently ignore

  if(w.isRevision) return;

  const a = {...w, title: w.title + ' (Part A)'};

  const b = {...w, title: w.title + ' (Part B)'};

  MODEL.weeks.splice(idx,1,a,b);

  recomputeDates();

  render();

}

 

function recomputeDates(){

  // Recompute week start dates from MODEL.startDate in order

  let cursor = startOfDay(MODEL.startDate);

  for(const w of MODEL.weeks){

    w.start = new Date(cursor);

    cursor = addDays(cursor,7);

  }

}

 

function setError(msg){

  const el = $('#err');

  if(!msg){ el.innerHTML=''; return;}

  el.innerHTML = `<div class="error" role="alert">${escapeHtml(msg)}</div>`;

  setTimeout(()=>{ el.innerHTML=''; }, 6000);

}

 

function escapeHtml(s){ return (s??'').toString().replace(/[&<>"']/g, m=>({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m])); }

 

/* ------------------ Wire up controls ------------------ */

$('#generateBtn').addEventListener('click', ()=>{

  const startStr = $('#startDate').value;

  const stream = $('#stream').value;

  const sitting = $('#sitting').value;

 

  if(!startStr){ setError('Please provide a study start date.'); return; }

 

  const startDate = new Date(startStr+'T00:00:00');

  const examDate = EXAMS[sitting];

 

  const out = buildSequence(startDate, examDate, stream);

  if(out.error){ setError(out.error); return; }

 

  MODEL.startDate = startDate;

  MODEL.examKey = sitting;

  MODEL.stream = stream;

  MODEL.weeks = out.weeks;

 

  $('#summary').hidden = false;

  $('#exit').hidden = true;

  recomputeDates();

  render();

});

 

$('#sitting').addEventListener('change', ()=>{

  // Regenerate immediately when exam month changes

  if(!MODEL.startDate) return; // not generated yet

  $('#generateBtn').click();

});

 

$('#resetBtn').addEventListener('click', ()=>{

  $('#startDate').value='';

  $('#stream').value='Both';

  $('#sitting').value='Jan';

  $('#summary').hidden = true;

  $('#exit').hidden = true;

  setError('');

  MODEL = {weeks:[], startDate:null, examKey:'Jan', stream:'Both'};

});

 

$('#startOver2').addEventListener('click', ()=>{

  $('#resetBtn').click();

  window.scrollTo({top:0,behavior:'smooth'});

});

  }

});
